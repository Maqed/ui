<TabsContent value="cli">

<Steps className="mb-0 pt-2">

<Step>Run the following command:</Step>

<CodeBlockCommand command="npx shadcn@latest add https://herocn.dev/r/sonner.json" />

<Step>Add the `Toaster` component to your root layout.</Step>

```tsx title="app/layout.tsx"
import { Toaster } from "@/components/ui/sonner"

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode
}>) {
  return (
    <html lang="en">
      <body>
        <main>{children}</main>
        <Toaster />
      </body>
    </html>
  )
}
```

</Steps>

<TabsContent value="manual">

<Steps className="mb-0 pt-2">

<Step>Install the following dependencies:</Step>

<CodeBlockCommand command="npm install sonner next-themes" />

<Step>Copy and paste the following code into your project.</Step>

<ComponentSource name="sonner" title="components/ui/sonner.tsx" />

<Step>Add the following css colors & utilities.</Step>

<RenderCSSShared
  light={["--surface", "--success", "--warning", "--default", "--default-foreground"]}
  dark={["--surface", "--warning", "--default", "--default-foreground"]}
  theme={["--color-surface", "--color-success", "--color-warning", "--color-default", "--color-default-foreground"]}
  css={["@utility pressible"]}
/>

<Step>Add the `Toaster` component to your root layout.</Step>

```tsx title="app/layout.tsx"
import { Toaster } from "@/components/ui/sonner"

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode
}>) {
  return (
    <html lang="en">
      <body>
        <Toaster />
        <main>{children}</main>
      </body>
    </html>
  )
}
```

</Steps>

```tsx
import { toast } from "sonner"
```

toast("Event has been created.")
